70 bool ErriezDS1302::begin()
71 {
72  // Initialize pins
73  DS1302_CLK_LOW();
74  DS1302_IO_LOW();
75  DS1302_CE_LOW();
76 
77  DS1302_CLK_OUTPUT();
78  DS1302_IO_OUTPUT();
79  DS1302_CE_OUTPUT();
80 
81  // Check zero bits in day week register
82  if (readRegister(DS1302_REG_DAY_WEEK) & 0xF8) {
83  return false;
84  }
85 
86  // Remove write protect
87  writeRegister(DS1302_REG_WP, 0);
88 
89  // Check write protect bit
90  if (readRegister(DS1302_REG_WP) & (1 << DS1302_BIT_WP)) {
91  // Error: RTC write protect bit not cleared
92  return false;
93  }
94 
95  return true;
96 }

120 void ErriezDS1302::clockEnable(bool enable)
121 {
122  uint8_t regSeconds;
123 
124  // Read seconds register
125  regSeconds = readRegister(DS1302_REG_SECONDS);
126 
127  // Set or clear HT bit
128  if (enable) {
129  // Clear to enable
130  regSeconds &= ~(1 << DS1302_BIT_CH);

132  // Set to disable

136  // Write seconds register
137  writeRegister(DS1302_REG_SECONDS, regSeconds);

196 bool ErriezDS1302::read(struct tm *dt)
197 {
198  uint8_t buffer[7];
199 
200  // Read clock date and time registers
201  transferBegin();
202  writeAddrCmd(DS1302_CMD_READ_CLOCK_BURST);
203  readBuffer(&buffer, sizeof(buffer));
204  transferEnd();
205 
206  // Clear dt
207  memset(dt, 0, sizeof(struct tm));
208 
209  // Convert BCD buffer to Decimal
210  dt->tm_sec = bcdToDec(buffer[0] & 0x7F);
211  dt->tm_min = bcdToDec(buffer[1] & 0x7F);
212  dt->tm_hour = bcdToDec(buffer[2] & 0x3f);
213  dt->tm_mday = bcdToDec(buffer[3] & 0x3F);
214  dt->tm_mon = bcdToDec(buffer[4] & 0x1f);
215  dt->tm_wday = bcdToDec(buffer[5] & 0x07);
216  dt->tm_year = bcdToDec(buffer[6]) + 100; // 2000-1900
217 
218  // Month: 0..11
219  if (dt->tm_mon) {
220  dt->tm_mon--;
221  }
222 
223  // Day of the week: 0=Sunday
224  if (dt->tm_wday) {
225  dt->tm_wday--;
226  }
227 
228  // Check buffer for valid data
229  if ((dt->tm_sec > 59) || (dt->tm_min > 59) || (dt->tm_hour > 23) ||
230  (dt->tm_mday < 1) || (dt->tm_mday > 31) || (dt->tm_mon > 11) || (dt->tm_year > 199) ||
231  (dt->tm_wday > 6))
232  {
233  return false;
234  }
235 
236  return true;
237 }

247 void ErriezDS1302::write(const struct tm *dt)
248 {
249  // Write clock registers(always 24H)
250  transferBegin();
251  writeAddrCmd(DS1302_CMD_WRITE_CLOCK_BURST);
252  writeByte(decToBcd(dt->tm_sec) & 0x7F); // Remove CH bit
253  writeByte(decToBcd(dt->tm_min) & 0x7F);
254  writeByte(decToBcd(dt->tm_hour) & 0x3F);
255  writeByte(decToBcd(dt->tm_mday) & 0x3F);
256  writeByte(decToBcd(dt->tm_mon + 1) & 0x1F);
257  writeByte(decToBcd(dt->tm_wday + 1) & 0x07);
258  writeByte(decToBcd(dt->tm_year % 100));
259  writeByte(0); // Including write protect = 0
260  transferEnd();
261 }

523 void ErriezDS1302::writeAddrCmd(uint8_t value)
524 {
525  // Write 8 bits to RTC
526  for (uint8_t i = 0; i < 8; i++) {
527  if (value & (1 << i)) {
528  DS1302_IO_HIGH();
529  } else {
530  DS1302_IO_LOW();
531  }
532  DS1302_PIN_DELAY();
533  DS1302_CLK_HIGH();
534  DS1302_PIN_DELAY();
535 
536  if ((value & (1 << DS1302_BIT_READ)) && (i == 7)) {
537  DS1302_IO_INPUT();
538  } else {
539  DS1302_CLK_LOW();
540  }
541  }
