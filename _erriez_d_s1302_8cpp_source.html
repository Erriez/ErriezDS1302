125  regOld = readClockRegister(DS1302_REG_SECONDS);

157 void DS1302::setDateTime(DS1302_DateTime *dateTime)
158 {
159  uint8_t ch;
160 
161  // Read CH bit
162  ch = (uint8_t)(readClockRegister(0) & (1 << DS1302_BIT_CH));
163 
164  // Write clock registers(always 24H)
165  transferBegin();
166  writeAddrCmd(DS1302_CMD_WRITE_CLOCK_BURST);
167  writeByte(ch | decToBcd((uint8_t)(dateTime->second & 0x7F)));
168  writeByte(decToBcd(dateTime->minute));
169  writeByte(decToBcd((uint8_t)(dateTime->hour & 0x3F)));
170  writeByte(decToBcd((uint8_t)(dateTime->dayMonth & 0x1F)));
171  writeByte(decToBcd((uint8_t)(dateTime->month & 0x1F)));
172  writeByte(decToBcd((uint8_t)(dateTime->dayWeek & 0x07)));
173  writeByte(decToBcd((uint8_t)((dateTime->year - 2000) & 0xFF)));
174  writeByte(0); // Including write protect = 0
175  transferEnd();
176 }

183 bool DS1302::getDateTime(DS1302_DateTime *dateTime)
184 {
185  uint8_t buf[7];
186 
187  // Read clock date and time registers
188  transferBegin();
189  writeAddrCmd(DS1302_CMD_READ_CLOCK_BURST);
190  readBuffer(&buf, sizeof(buf));
191  transferEnd();
192 
193  // Convert BCD buffer to Decimal
194  dateTime->second = bcdToDec(buf[0] & 0x7f); // Without CH bit from seconds register
195  dateTime->minute = bcdToDec(buf[1]);
196  dateTime->hour = bcdToDec(buf[2]);
197  dateTime->dayMonth = bcdToDec(buf[3]);
198  dateTime->month = bcdToDec(buf[4]);
199  dateTime->dayWeek = bcdToDec(buf[5]);
200  dateTime->year = 2000 + bcdToDec(buf[6]);
201 
202  // Check buffer for valid data
203  if ((dateTime->second > 59) ||
204  (dateTime->minute > 59) ||
205  (dateTime->hour > 23) ||
206  (dateTime->dayMonth < 1) || (dateTime->dayMonth > 31) ||
207  (dateTime->month < 1) || (dateTime->month > 12) ||
208  (dateTime->dayWeek < 1) || (dateTime->dayWeek > 7) ||
209  (dateTime->year > 2099))
210  {
211  memset(dateTime, 0x00, sizeof(DS1302_DateTime));
212  return false;
213  }
214 
215  return true;
216 }

397 void DS1302::writeAddrCmd(uint8_t value)
398 {
399  // Write 8 bits to RTC
400  for (uint8_t i = 0; i < 8; i++) {
401  if (value & (1 << i)) {
402  DS1302_IO_HIGH();
403  } else {
404  DS1302_IO_LOW();
405  }
406  DS1302_PIN_DELAY();
407  DS1302_CLK_HIGH();
408  DS1302_PIN_DELAY();
409 
410  if ((value & (1 << DS1302_BIT_READ)) && (i == 7)) {
411  DS1302_IO_INPUT();
412  } else {
413  DS1302_CLK_LOW();
414  }
415  }
416 }
